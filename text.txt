  <h2>Limite de até 262 caracteres</h2>

  <textarea id="texto" maxlength="262" oninput="atualizarContagem()" placeholder="Digite seu texto aqui (máx. 262 caracteres)..."></textarea>
  <p>Caracteres: <span id="contador">0</span> / 262</p>
  <p id="aviso"></p>

  <script>
    function atualizarContagem() {
      const campo = document.getElementById("texto");
      const contador = document.getElementById("contador");
      const aviso = document.getElementById("aviso");

      const comprimento = campo.value.length;
      contador.textContent = comprimento;

      if (comprimento >= 262) {
        aviso.textContent = "⚠️ Você atingiu o limite máximo de 262 caracteres!";
      } else {
        aviso.textContent = "";
      }
    }
  </script>
